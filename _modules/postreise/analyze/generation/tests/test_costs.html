[docs]def test_calculate_cost_equal_both_methods(mock_gencost, mock_pg, mock_scenario):

[docs]def test_pass_too_many_things(mock_gencost, mock_pg, mock_scenario):
    with pytest.raises(ValueError) as excinfo:
        calculate_costs(gencost=mock_gencost, pg=mock_pg, scenario=mock_scenario)
    assert "Either scenario XOR (pg AND gencost) must be specified" in str(
        excinfo.value
    )




[docs]def test_negative_pg(mock_gencost, mock_pg):
    with pytest.raises(ValueError) as excinfo:
        calculate_costs(gencost=mock_gencost, pg=-1 * mock_pg.iloc[:, 2])
