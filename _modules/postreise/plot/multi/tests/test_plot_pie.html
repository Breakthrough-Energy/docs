[docs]def test_roll_up_small_pie_wedges_with_zero_values():
    ax_data, labels = _roll_up_small_pie_wedges({"coal": 0, "solar": 1, "hydro": 5}, 0)
    assert ax_data == {"solar": 1, "hydro": 5}
    assert labels == ["Solar", "Hydro"]
