            ("base_hydro", base_hydro),
            ("base_solar", ""),
            ("base_wind", ""),
            ("change_table", False),

            ("end_date", "2016-12-31 23:00:00"),
            ("interval", ""),
            ("engine", ""),
            ("runtime", ""),
            ("infeasibilities", ""),
        ]
    )
