[docs]def get_resource(model):
    """Return arrrangement of generator types.

    :param str model: grid model
    """
    _lookup = {
        "usa_tamu": USAResource,
        "hifld": USAResource,
        "europe_tub": EUResource,
    }
    return _lookup[model]().__dict__
