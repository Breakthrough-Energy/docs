[docs]def test_max_times_reached():
    @retry(max_attempts=3, allowed_exceptions=CustomError)
    def no_fail(x=[]):
        x.append(len(x))
        raise CustomError()

    counts = []
    no_fail(counts)
    assert len(counts) == 3

[docs]def test_counter_attribute():
    limit = 3

    @retry(max_attempts=limit, allowed_exceptions=CustomError)
    def best_attempt():
        x = best_attempt.retry_count
        if best_attempt.retry_count < limit:
            raise CustomError
        return x

    assert limit == best_attempt()
